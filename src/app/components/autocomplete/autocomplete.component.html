Autocomplete link


Atributos 


    modelValue: Pega o valor selecionado.

    list: Lista de opções para o select.

    loading: Utilizado para controlar a barra de carregamento do select.

    select: Dispara um evento após a seleção de um item.

Simples

HTML 

    <ui-input-container>
      <input type="text"
             [(ngModel)]="searchTeams"
             [(modelValue)]="teste"
             uiInput
             uiAutocomplete
             (input)="search()"
             (focus)="search()"
             [list]="teamsFiltrados"
             [loading]="loading"
             [select]="finish.bind(this)">
      <label>Clubes</label>
    </ui-input-container>

JS 

    search() {
        clearInterval(this.timing);
        this.loading = true;
        this.timing = setTimeout(() => {
          this.loading = false;
          if (!this.searchTeams) {
            this.teamsFiltrados = this.teams;
            return;
          }
          this.teamsFiltrados = this.teams.filter(item => {
            return (item.indexOf(this.searchTeams) !== -1);
          });
        }, 300);
      }

    finish(item) {
        console.log(item); // Mostra o item selecionado
    }
